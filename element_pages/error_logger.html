Error Logger
Explanation
This element provides a class to save or check for engine errors.
Also you can use for your things.
It's a simple queue, you add error or get & delete de last inserted by a predefined structure.
Error struct 
        
  class ErrorLog
  {
    class Error
    {
    public:
      class string function_;
      class string error_;
      class string line_;

      Error();
      Error(const Error &other);
      Error(const std::string error, const std::string function = "\0", const std::string line = "\0");

      ~Error();

      Error &operator=(const Error &other);
    };
  }
Functions 
        
  class ErrorLog
  {
    ErrorLog();
    ~ErrorLog();

    void addError(class string error = "\0", 
                class string function = "\0", 
                class string line = "\0");
    struct Error getError();
  }

  class JAM_Engine
  {
    static void AddError(class string error = "\0", 
                         class string function = "\0", 
                         class string line = "\0");
    static struct Error GetError();
  }
Example
To use the error logger you only need to call the engine functions to add or get it. 
        
  #include <jam_engine.h>
  void main() 
  {
    JAM_Engine::AddError("error", __FUNCTION__, "72");
    struct Error error = JAM_Engine::GetError();

    printf(stdout, "Error: %s, Function: %s, Line: %s", error.name, error.function, error.line);
    // May print: Error: error, Function: main, Line: 72
  }
